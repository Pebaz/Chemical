lass Peekable(it):
   def __init__(self, items):
       it.__init__(self, items)
       self.ahead = None
       self.done = False
   def peek(self):
       if self.done:
           raise StopIteration()
       if not self.ahead:
           self.ahead = next(self.items)
       return self.ahead
   def __next__(self):
       if self.done:
           raise StopIteration()

           if not self.ahead:
               self.ahead = next(self.items)
       except StopIteration:
           ...
       ret = self.ahead

           self.ahead = next(self.items)
       except StopIteration:
           self.done = True
           return ret
       return ret

ef max_by(self, closure):
   max_val = self.next()

           max_val = i

   return max_val

ef eq_by(self, other, closure):
   other = it(other)
   while True:
       try:
           a = next(self)
       except StopIteration:
           a = StopIteration
       try:
           b = next(other)
       except StopIteration:
           b = StopIteration
       if a == StopIteration and b == StopIteration:
           return True
       elif (a, b) == (StopIteration, b) or (a, b) == (a, StopIteration):
           return False
       elif not closure(a, b):
           return False

ef flatten(self):
   links = it()
